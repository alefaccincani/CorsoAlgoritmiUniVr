\section{Divide et Impera}

\begin{softnote}
	Il \textbf{divide et impera} è una delle tecniche algoritmiche
	più potenti. L'idea è semplice: se un problema è troppo grande
	per essere risolto direttamente, spezzalo in pezzi più piccoli,
	risolvi i pezzi, e combina le soluzioni.
\end{softnote}
\subsection{Idea Generale}

Il paradigma divide et impera prevede tre passi:
\begin{enumerate}
	\item \textbf{Divide:} suddividi il problema in sottoproblemi
	di dimensione inferiore;
	\item \textbf{Impera:} risolvi ricorsivamente i sottoproblemi
	(quando la dimensione è abbastanza piccola, risolvi
	direttamente);
	\item \textbf{Combina:} unisci i risultati per ottenere
	la soluzione al problema originale.
\end{enumerate}

\begin{nota}
	\textbf{Analogia:} immagina di dover ordinare un mazzo di 52 carte.
	Invece di ordinarle tutte in una volta (difficile), le dividi in
	due mazzi da 26, ordini ciascun mazzo separatamente, e poi fondi
	i due mazzi ordinati. Questo è esattamente MergeSort!
\end{nota}

\subsubsection*{Aspetti cruciali}

\begin{itemize}
	\item \textbf{Suddivisione:} deve produrre sottoproblemi il più
	possibile \textit{disgiunti}.
	\item \textbf{Ricorsione:} deve terminare quando la dimensione
	permette una risoluzione diretta (caso base).
	\item \textbf{Combinazione:} non deve richiedere un tempo
	maggiore di quello della risoluzione dei sottoproblemi.
\end{itemize}

% ----------------------------------------------------------------
\subsection{Schema Generale}

\begin{algorithm}[H]
	\caption{DivideEtImpera($X$)}
	\KwIn{$X$: istanza di dimensione $|X| = n$}
	\KwOut{$Y$: soluzione}
	\If{$|X| \le n_0$}{
		\Return{adhoc($X$)}\tcp*{Caso base: soluzione diretta}
	}
	Dividi $X$ in sottoistanze $X_1, X_2, \ldots, X_a$\;
	\For{$i = 1$ \KwTo $a$}{
		$Y_i \gets \text{DivideEtImpera}(X_i)$\tcp*{Ricorsione}
	}
	$Y \gets \text{combina}(Y_1, \ldots, Y_a)$\;
	\Return{$Y$}\;
\end{algorithm}

\begin{softnote}
	\begin{itemize}
		\item \textbf{adhoc}($X$): risolve istanze piccole in tempo
		costante o lineare.
		\item \textbf{dividi}: produce $a$ sottoistanze di dimensione
		$n/b$ ciascuna ($b>1$).
		\item Se $a=1$: si chiama \textit{semplificazione} (es.\ ricerca binaria).
		\item \textbf{combina}: ha solitamente complessità $O(n^d)$.
	\end{itemize}
\end{softnote}

% ----------------------------------------------------------------
\subsection{Costo Computazionale}

Il costo è la soluzione di un'\textbf{equazione di ricorrenza}:

\[
T(n) =
\begin{cases}
	g(n)           & n \le n_0 \\
	a\,T(n/b)+f(n) & \text{altrimenti}
\end{cases}
\]

dove $f(n) = \text{tempo(dividi)} + \text{tempo(combina)}$.

\begin{nota}
	\textbf{Ricorsione o iterazione?}
	\begin{itemize}
		\item \textbf{Pro iterazione:} elimina l'overhead delle chiamate
		ricorsive; risparmia fino a $\Theta(h)$ spazio ausiliario.
		\item \textbf{Contro:} l'algoritmo diventa più complicato da scrivere.
	\end{itemize}
\end{nota}

\subsubsection{Problema Prodotto — Algoritmo di Karatsuba}

\begin{softnote}
	Il primo esempio concreto di divide et impera è la moltiplicazione
	di interi grandi. La moltiplicazione tradizionale è $O(n^2)$:
	possiamo fare meglio?
\end{softnote}

\begin{definizione}[Problema del Prodotto]
	\textbf{Input:} due interi $x$, $y$ di $n$ bit ciascuno.\\
	\textbf{Output:} il loro prodotto $x \cdot y$.
\end{definizione}

Il costo del prodotto tradizionale (moltiplicazione in colonna) è $O(n^2)$.

\paragraph{Idea divide et impera.}
Si decompongono $x$ e $y$ nelle loro metà alta e bassa:
\[
x = 2^{n/2}\,x_H + x_L, \qquad y = 2^{n/2}\,y_H + y_L.
\]
Il prodotto diventa:
\[
x \cdot y = 2^n\,x_H y_H
+ 2^{n/2}(x_H y_L + x_L y_H)
+ x_L y_L.
\]
Questa decomposizione usa \textbf{4 moltiplicazioni} di dimensione $n/2$,
quindi $T(n) = 4T(n/2) + \Theta(n) = \Theta(n^2)$: nessun miglioramento.

\paragraph{Osservazione di Gauss (Karatsuba).}
Con un trucco analogo a quello usato per i numeri complessi,
si può calcolare il termine misto $x_H y_L + x_L y_H$
usando \textbf{una sola} moltiplicazione aggiuntiva:
\[
x_H y_L + x_L y_H
= (x_H + x_L)(y_H + y_L) - x_H y_H - x_L y_L.
\]
Poiché $x_H y_H$ e $x_L y_L$ sono già stati calcolati, bastano
\textbf{3 moltiplicazioni} in totale.

\begin{algorithm}[H]
	\caption{Prodotto$(x, y, n)$}
	\KwIn{Interi $x$, $y$ di $n$ bit}
	\KwOut{$x \cdot y$}
	\If{$n = 1$}{\Return $x \cdot y$\;}
	$x_H \gets \lfloor x / 2^{n/2} \rfloor$;\quad
	$x_L \gets x \bmod 2^{n/2}$\;
	$y_H \gets \lfloor y / 2^{n/2} \rfloor$;\quad
	$y_L \gets y \bmod 2^{n/2}$\;
	$p \gets \text{Prodotto}(x_H,\; y_H,\; n/2)$\;
	$q \gets \text{Prodotto}(x_L,\; y_L,\; n/2)$\;
	$r \gets \text{Prodotto}(x_H + x_L,\; y_H + y_L,\; n/2)$\;
	\Return $2^n p + 2^{n/2}(r - p - q) + q$\;
\end{algorithm}

\paragraph{Analisi.}
\[
T(n) = 3\,T(n/2) + \Theta(n).
\]
Applicando il Teorema dell'Esperto (caso 1, con
$\log_2 3 \approx 1.585 > 1$):
\[
\boxed{T(n) = \Theta(n^{\log_2 3}) \approx \Theta(n^{1.585})}.
\]

\begin{nota}
	$\Theta(n^{1.585})$ è significativamente meglio di $\Theta(n^2)$
	del metodo tradizionale. Per $n = 1000$ bit, il rapporto è circa
	$1000^{0.415} \approx 50\times$ più veloce.
\end{nota}

\subsubsection{Prodotto di Interi Grandi}

\begin{definizione}[Problema Prodotto]
	\textbf{Input:} due interi $x$, $y$ di $n$ bit ciascuno ($n = 2^k$).\\
	\textbf{Output:} il loro prodotto $xy$.\\
	\textbf{Costo naïve:} $O(n^2)$.
\end{definizione}

Si suddivide $x = 2^{n/2}x_L + x_R$, $y = 2^{n/2}y_L + y_R$:
\[
xy = 2^n(x_L y_L) + 2^{n/2}(x_L y_R + x_R y_L) + x_R y_R
\]

Questo richiede 4 moltiplicazioni di dimensione $n/2$, dando
$T(n) = 4T(n/2) + O(n) = O(n^2)$ — nessun miglioramento.

\begin{hardnote}
	\textbf{Trucco di Gauss} (1777--1855): bastano \textbf{3 moltiplicazioni}:
	\[
	P_1 = x_L y_L,\quad P_2 = x_R y_R,\quad
	P_3 = (x_L+x_R)(y_L+y_R)
	\]
	\[
	xy = 2^n P_1 + 2^{n/2}(P_3 - P_1 - P_2) + P_2
	\]
\end{hardnote}

\begin{algorithm}[H]
	\caption{Prodotto($x$, $y$)}
	\KwIn{$x$, $y$ interi non negativi di $n = 2^\ell$ bit}
	\KwOut{prodotto $xy$}
	\If{$n \le 64$}{\Return{$x \cdot y$}\tcp*{Caso base $\Theta(1)$}}
	$x_L, x_R \gets$ metà sinistra/destra di $x$\tcp*{Divide $\Theta(n)$}
	$y_L, y_R \gets$ metà sinistra/destra di $y$\;
	$P_1 \gets \text{Prodotto}(x_L,\,y_L)$\;
	$P_2 \gets \text{Prodotto}(x_R,\,y_R)$\tcp*{Impera $3T(n/2)$}
	$P_3 \gets \text{Prodotto}(x_L+x_R,\,y_L+y_R)$\;
	\Return{$(P_1 \ll n) + ((P_3-P_1-P_2) \ll n/2) + P_2$}
	\tcp*{Combina $\Theta(n)$}
\end{algorithm}

\begin{softnote}
	\textbf{Complessità:}
	\[
	T(n) = 3\,T(n/2) + \Theta(n) = \Theta(n^{\log_2 3})
	= \Theta(n^{1.585})
	\]
\end{softnote}

\subsubsection{MergeSort}

Lo schema divide et impera per MergeSort:
\begin{itemize}
	\item \textbf{Divide:} 2 sottoarray di $n/2$ elementi — costo $\Theta(1)$.
	\item \textbf{Impera:} ordina i due sottoarray ricorsivamente.
	\item \textbf{Combina:} fondi i due sottoarray ordinati — costo $\Theta(n)$.
\end{itemize}

\subsubsection*{Analisi di complessità}

\[
T(n) =
\begin{cases}
	\Theta(1)             & n = 1 \\
	2\,T(n/2) + \Theta(n) & n > 1
\end{cases}
\;\Rightarrow\; T(n) = \Theta(n \log n)
\]

\subsubsection{Massimo SottoArray}

\begin{definizione}[Massimo SottoArray]
	\textbf{Input:} un array di interi $A[1:n]$.\\
	\textbf{Output:} il sottoarray non vuoto $A[i:j]$ con somma massima:
	\[
	\max_{1 \le i \le j \le n} \sum_{k=i}^{j} A[k]
	\]
\end{definizione}

\begin{esempio}
	Profitti 2001--2012: il massimo sottoarray è 2008--2010 con somma
	$32 - 5 + 10 = 37$.
\end{esempio}

\subsubsection*{Strategia divide et impera}

Posto $mid = \lfloor(low+high)/2\rfloor$, la soluzione è in uno di tre casi:
\begin{enumerate}
	\item Interamente in $A[low:mid]$
	\item Interamente in $A[mid+1:high]$
	\item \textbf{A cavallo} di $mid$
\end{enumerate}

\begin{algorithm}[H]
	\caption{FindMaxCrossingSubarray($A$, $low$, $mid$, $high$)}
	\KwIn{Array $A$, indici $low \le mid < high$}
	\KwOut{$(i^*, j^*, \text{somma})$}
	$\mathit{leftSum} \gets -\infty$\; $\mathit{sum} \gets 0$\;
	\For{$i \gets mid$ \textbf{downto} $low$}{
		$\mathit{sum} \gets \mathit{sum} + A[i]$\;
		\If{$\mathit{sum} > \mathit{leftSum}$}{
			$\mathit{leftSum} \gets \mathit{sum}$\; $i^* \gets i$\;
		}
	}
	$\mathit{rightSum} \gets -\infty$\; $\mathit{sum} \gets 0$\;
	\For{$j \gets mid+1$ \textbf{to} $high$}{
		$\mathit{sum} \gets \mathit{sum} + A[j]$\;
		\If{$\mathit{sum} > \mathit{rightSum}$}{
			$\mathit{rightSum} \gets \mathit{sum}$\; $j^* \gets j$\;
		}
	}
	\Return{$(i^*,\; j^*,\; \mathit{leftSum}+\mathit{rightSum})$}\;
\end{algorithm}

\begin{algorithm}[H]
	\caption{FindMaximumSubarray($A$, $low$, $high$)}
	\KwIn{Array $A[low:high]$}
	\KwOut{$(i^*,j^*,\text{somma massima})$}
	\If{$high = low$}{\Return{$(low,\,high,\,A[low])$}\tcp*{Caso base}}
	$mid \gets \lfloor(low+high)/2\rfloor$\;
	$(i_L, j_L, s_L) \gets \text{FindMaximumSubarray}(A, low, mid)$\;
	$(i_R, j_R, s_R) \gets \text{FindMaximumSubarray}(A, mid+1, high)$\;
	$(i_C, j_C, s_C) \gets \text{FindMaxCrossingSubarray}(A,low,mid,high)$\;
	\Return{$\arg\max\{s_L, s_R, s_C\}$}\;
\end{algorithm}

\[
T(n) = 2\,T(n/2) + \Theta(n) \;\Rightarrow\; T(n) = \Theta(n \log n)
\]

\subsection{Equazioni di Ricorrenza}

\begin{definizione}[Equazione di ricorrenza algoritmica]
	\[
	T(n) =
	\begin{cases}
		K             & n \le k \\
		a\,T(n/b)+C(n)& \text{altrimenti}
	\end{cases}
	\]
\end{definizione}

% --------------------------------
\subsubsection{Metodo di Sostituzione}

\begin{softnote}
	\textbf{(1) ipotizza} una soluzione; \textbf{(2) dimostra}
	per induzione che è corretta.
\end{softnote}

\begin{esempio}
	Risolvere $T(n) = 2T\!\left(\lfloor n/2 \rfloor\right) + n$.
	
	\textbf{Ipotesi:} $T(n) \le cn\log n$. Passo induttivo:
	\begin{align*}
		T(n) &\le 2\bigl(c\lfloor n/2\rfloor \log\lfloor n/2\rfloor\bigr) + n \\
		&\le cn\log(n/2) + n = cn\log n - cn + n \le cn\log n
	\end{align*}
	valido per $c \ge 1$. \checkmark
\end{esempio}

\begin{hardnote}
	\textbf{Errore comune:} se ipotizzo $T(n) = O(n)$ e ottengo
	$T(n) \le cn + n = (c+1)n$, \textbf{non} ho dimostrato $T(n) \le cn$.
	La soluzione $O(n)$ è quindi errata per questa ricorrenza.
\end{hardnote}

% --------------------------------
\subsubsection{Metodo dell'Albero di Ricorsione}

\begin{esempio}
	Risolvere $T(n) = 3T(\lfloor n/4 \rfloor) + \Theta(n^2)$.
	
	Costo livello $i$: $3^i \cdot c(n/4^i)^2 = (3/16)^i\,cn^2$.
	
	Totale (serie geometrica con ragione $3/16 < 1$):
	\[
	\sum_{i=0}^{\log_4 n}(3/16)^i\,cn^2
	< \frac{cn^2}{1-3/16} = \frac{16}{13}cn^2 = O(n^2)
	\]
\end{esempio}

% --------------------------------
\subsubsection{Metodo dell'Esperto}

\begin{teorema}[Teorema dell'Esperto]
	Date costanti $a>0$, $b>1$ e $f(n)$ asintoticamente positiva,
	per $T(n) = aT(n/b) + f(n)$:
	
	\begin{enumerate}
		\item Se $f(n) = O(n^{(\log_b a)-\varepsilon})$:
		$\;T(n) = \Theta(n^{\log_b a})$
		\hfill\textit{(foglie dominano)}
		
		\item Se $f(n) = \Theta(n^{\log_b a}\log^k n)$:
		$\;T(n) = \Theta(n^{\log_b a}\log^{k+1} n)$
		\hfill\textit{(bilanciato)}
		
		\item Se $f(n) = \Omega(n^{(\log_b a)+\varepsilon})$
		e $af(n/b)\le cf(n)$:
		$\;T(n) = \Theta(f(n))$
		\hfill\textit{(radice domina)}
	\end{enumerate}
\end{teorema}

\begin{esempio}
	\textbf{Caso 1 —} $T(n) = 9T(n/3) + n$:
	$n^{\log_3 9} = n^2$, $f(n)=n=O(n^{2-1})$ $\Rightarrow T(n)=\Theta(n^2)$.
\end{esempio}

\begin{esempio}
	\textbf{Caso 2 —} $T(n) = T(n/2) + 1$ (ricerca binaria):
	$n^{\log_2 1}=1$, $f(n)=\Theta(1)$ $\Rightarrow T(n)=\Theta(\log n)$.
\end{esempio}

\begin{esempio}
	\textbf{Caso 3 —} $T(n) = 3T(n/4) + n^2$:
	$n^{\log_4 3}\approx n^{0.79}$, $f(n)=n^2$, $3(n/4)^2 \le \tfrac{3}{16}n^2$
	$\Rightarrow T(n)=\Theta(n^2)$.
\end{esempio}

\begin{hardnote}
	\textbf{Errore comune nel metodo di sostituzione.}
	Si consideri $T(n) = 2T(\lfloor n/2 \rfloor) + n$ e si ipotizzi
	$T(n) = O(n)$, cioè $T(n) \le cn$.
	Passo induttivo:
	\[
	T(n) \le 2c\lfloor n/2 \rfloor + n \le cn + n = (c+1)n.
	\]
	Questo \textbf{non} dimostra $T(n) \le cn$, perché la costante
	è aumentata da $c$ a $c+1$. L'ipotesi $O(n)$ è quindi \textbf{errata}
	per questa ricorrenza; la soluzione corretta è $T(n) = O(n \log n)$.
\end{hardnote}

\begin{osservazione}
	Il Teorema dell'Esperto \textbf{non si applica} quando $f(n)$ non è
	polinomialmente separabile da $n^{\log_b a}$. Esempio tipico:
	\[
	T(n) = 2T(n/2) + n\log n.
	\]
	Qui $f(n) = n\log n$ e $n^{\log_2 2} = n$, ma
	$n\log n / n = \log n$ non è $\Omega(n^\varepsilon)$ per nessun
	$\varepsilon > 0$: non si rientra in nessuno dei tre casi.
	La soluzione è $T(n) = \Theta(n \log^2 n)$, ottenibile con l'albero
	di ricorsione o con il metodo di sostituzione.
\end{osservazione}

\begin{hardnote}
	\textbf{Caso non applicabile:} $T(n) = 2T(n/2) + n\log n$.
	
	$n^{\log_2 2}=n$. $f(n)=n\log n$ è più grande di $n$ ma non
	\textit{polinomialmente} più grande: non esiste $\varepsilon>0$
	con $n\log n = \Omega(n^{1+\varepsilon})$. Occorre il metodo
	di sostituzione.
\end{hardnote}
